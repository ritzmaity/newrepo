package org.example;

public class Switch_Case {
    public static void main(String[] args) {
        int value = 1;
        if (value == 1){
            System.out.println("value is 1");
        } else if (value == 2) {
            System.out.println("value is 2");
        } else {
            System.out.println("Neither 1 nor 2");
        }
    }
}
